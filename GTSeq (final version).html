Introduction

We start with vcf, in which all genotypes with fewer than 8 reads or with GQ < 20 were set to missing and then all SNPs with > 50% missing data were filtered out.

Select SNPs and alleles of interest

Calculate allele frequencies, number of alleles, and global MAF for each SNP

#vector of allele frequencies converted into df
af <- apply(tab(g, freq=TRUE), 2, mean, na.rm=TRUE)
af_df <- data.frame(allele = names(af), freq = af) %>% 
  mutate(marker = str_remove(allele, "_[0-9]+_[0-9]+[.].+$"),
         pos = as.numeric(str_extract(allele, "_([0-9]+)_[0-9]+[.].+$", group = 1)))
row.names(af_df) <- NULL

#number of alleles and MAF for each marker
af_info <- af_df  %>% 
  group_by(marker, pos) %>% summarise(Nall = n(),
                                      MAF = 1 - max(freq))
row.names(af_info) <- NULL

Identify SNP with the highest MAF for each marker, select only markers with MAF >=0.1, identify the most common allele for each SNP, plot fraction of missing data.

#SNP with the highest MAF for each marker
af_sum <- af_info %>% group_by(marker) %>% slice_max(order_by = MAF, n = 1, with_ties = FALSE)

#markers with three-allelic SNPs, these can be handled by pooling the two rarer alleles, but as there are only a few of them, I decided to drop them for the moment
three_all <- af_sum %>% filter(Nall == 3)

Population based analyses

Adding metainfo and calculating sample sizes

Get populations for individuals and add this info to genind. Note, that populations are based on the information in bam names. Additionally, the translation table contains information about the three populations for which pgdh allele frequencies in both autumn and spring were available (locality and seasona - for the remaining populations set toNA`).

#translation table of population names
poptr <- read_tsv("data/pop_translation_table.txt")


#get individual names as data frame
s <- data.frame(ID = indNames(gind)) %>% 
  mutate(temp = str_extract(ID, "sorted.(.+)_001.bam", group = 1),
         sample = str_extract(temp, "-([0-9]+)$", group = 1),

  left_join(poptr, by = "pop_orig") %>%

Hardy-Weinberg

Hardy-Weinberg tests for populations with sample sizes > 15

min15 <- pop_ss %>% filter(N > 15) %>% pull(pop)
HWE <- lapply(min15, function(x){
  p <- gind[pop(gind) == x, ]
  hw <- hw.test(p, B = 1000)
  hwdf <- data.frame(pop = x, SNP = row.names(hw), Pchi2 = hw[, 3], Pex = hw[, 4])
  row.names(hwdf) <- NULL
  return(hwdf)}) %>% bind_rows()

Allele frequency differences between season, we have only three pairs of populations.

sp <- PCAdf_seas %>% pull(pop) %>% unique()

#genind with pops of interest only
gind_seas <- gind[pop(gind) %in% sp,]

ids_seas <- data.frame(sample = indNames(gind_seas)) %>% left_join(PCAdf_seas, by = "sample") %>% 
  mutate(loc_seas = paste(locality, season, sep = "_"))

pop(gind_seas) <- ids_seas$loc_seas

pop_af <- lapply(ids_seas$loc_seas %>% unique(), function(x){
  p <- gind_seas[pop(gind_seas) == x, ]
  pf <- apply(tab(p, freq = TRUE), 2, mean, na.rm = TRUE)
  pdf <- data.frame(pop = x, allele = names(pf), freq = pf)
  row.names(pdf) <- NULL
  return(pdf)
  }) %>% bind_rows()

seas_pop_alleles <- pop_af %>% filter(allele %in% alleles) %>% mutate(MAF = 1-freq) %>% 
  mutate(pop_id = pop,
         pop = str_remove(pop_id, "_.+"),
         season = str_extract(pop_id, "_(.+)", group = 1)) %>% 
  select(pop, season, allele, MAF) %>% 
  pivot_wider(names_from = season, values_from = MAF) %>% 
  mutate(delta = abs(autumn - spring),
         meanMAF = (autumn + spring) / 2)

Conclusions

  • there’s very little genetic differentiation between localities
  • for sure, pgdh allele frequency changes between season are not larger than for other markers; I don’t think we need a more sophisticated analysis at this point
